class Solution {
public:
    int minimumObstacles(vector<vector<int>>& grid) {
        int dir[5]={-1, 0, 1, 0, -1}, n=grid.size(), m=grid[0].size();
        queue<pair<int, int>> q;
        vector<vector<int>> DP(n, vector<int>(m, INT_MAX-2));
        DP[0][0] = 0;
        q.push({0, 0});
        while(!q.empty()){
            pair<int, int> now = q.front();
            q.pop();
            for(int i=0; i<4; ++i){
                int y=now.first+dir[i], x=now.second+dir[i+1];
                if(y>=0&&y<n&&x>=0&&x<m){
                    int dep = DP[now.first][now.second]+grid[y][x];
                    if(dep<DP[y][x]){
                        DP[y][x]=dep;
                        q.push({y, x});
                    }
                }
            }
        }
        return DP[n-1][m-1];
    }
};

class Solution {
public:
    int minimumObstacles(vector<vector<int>>& grid) {
        int dir[5]={-1, 0, 1, 0, -1}, n=grid.size(), m=grid[0].size();
        queue<pair<int, int>> q, q2;
        int DP[n][m];
        memset(DP, -1, sizeof(DP));
        DP[0][0] = grid[0][0];
        q.push({0, 0});
        while(!q.empty()){
            pair<int, int> now = q.front();
            q.pop();
            for(int i=0; i<4; ++i){
                int y=now.first+dir[i], x=now.second+dir[i+1];

                    DP[y][x] = DP[now.first][now.second]+grid[y][x];
                    if(grid[y][x]) q2.push({y, x});
                    else q.push({y, x});
                }
            }
            if(q.empty()) swap(q, q2);
        }
        return DP[n-1][m-1];
    }
};
